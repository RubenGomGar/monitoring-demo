### 1. Add Request && Limits Deployment

```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: aspire-api-deployment
  namespace: aspire
spec:
  replicas: 1
  selector:
    matchLabels:
      app: aspire-api
  template:
    metadata:
      labels:
        app: aspire-api
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8080"
        prometheus.io/path: "/metrics" 
    spec:
      containers:
      - name: aspire-api
        image: rlggacr.azurecr.io/aspire-api:latest
        ports:
        - containerPort: 8080
        resources:
          limits:
            memory: 800Mi
            cpu: 250m
          requests:
            memory: 800Mi
            cpu: 250m
      imagePullSecrets:
      - name: acr-secret
```

### 2. Configure HPA


```yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: aspire-api-hpa
  namespace: aspire
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: aspire-api-deployment
  minReplicas: 1
  maxReplicas: 5
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 50
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 70
```
